# 环境变量

echo $PATH

## 用户环境变量

vim     ~/.bashrc
末尾加上export PATH=newpath:$PATH  
source  ~/.bashrc

export PATH=newpath:$PATH  
立即生效，临时的环境变量  

## 系统环境变量

vi     /etc/profile  

末尾加上：
export ABC=path
export PATH=newpath:$PATH

source /etc/profile
