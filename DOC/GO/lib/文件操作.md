# 文件操作

## 文件整体读写

```
import (
	"fmt"
	"io/ioutil"
)

//读文件
b, err := ioutil.ReadFile(filepathname)
if err != nil {
	fmt.Println("ioutil ReadFile error: ", err)
	return
}
fmt.Println(string(b))

//写文件
err := ioutil.WriteFile(filepathname, ([]byte)("x,y\na,b"), 0666)
if err != nil {
	fmt.Println("ioutil WriteFile error: ", err)
}

```

## 按行读写文件

```
import (
	"bufio"
	"fmt"
	"io"
	"os"
)
```

+ 读文件
```
f, err := os.Open("D:\\1.txt")
if err != nil {
	fmt.Println("os Open error: ", err)
	return
}
defer f.Close()

f, err := os.Open("D:\\a.rc")
if err != nil {
	fmt.Println("os Open error: ", err)
	return
}
defer f.Close()
br := bufio.NewReader(f)
for {
	line, _, err := br.ReadLine()
	if err == io.EOF {
		break
	} else {
		if err != nil {
			fmt.Println("br ReadLine error: ", err)
			return
		}
		fmt.Println(string(line))
	}
}
```

+ 写文件
```
f, err := os.Create("D:\\1.txt")
if err != nil {
	fmt.Println("os Open error: ", err)
	return
}
defer f.Close()
content := map[string]string{
	"a": "a",
	"b": "b",
}
bw := bufio.NewWriter(f)
for k, v := range content {
	bw.WriteString(k + "," + v + "\n")
}
bw.Flush()
```

## 追加内容

```
f, err := os.OpenFile("D:\\1.txt", os.O_RDWR|os.O_CREATE|os.O_APPEND, 0777)
if err != nil {
    fmt.Println("os OpenFile error: ", err)
    return
}
defer f.Close()

f.WriteString("e,e")
```
