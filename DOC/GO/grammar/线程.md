## 通道

+ 无缓冲通道
```
sig := make(chan int)

go func() {
	var x int
	fmt.Scan(&x)
	sig <- x //输入信号
}()

<-sig //等待(取出)信号
fmt.Println("end")

close(sig)
```

+ 有缓冲通道
```
ch := make(chan int, 10)//缓冲区大小为10个int
go func() {
	for {
		var x int
		fmt.Scan(&x)
		ch <- x //当缓冲区满时，等待读取缓冲区
		fmt.Printf("chCount:%d\n", len(ch)) //缓冲区当前包含的数据数量
	}
}()


go func() {
	for {
		fmt.Printf("w:%d\n", <-ch)
		time.Sleep(time.Duration(3) * time.Second)
	}
}()
```
