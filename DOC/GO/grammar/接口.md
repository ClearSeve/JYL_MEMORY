# 接口

## 定义接口
```
//INTER 接口类型
type INTER interface {
	funadd(val int) int
	funmul() int
}
```

## 实现接口
```
//AddStu  可用转换为接口的结构体（分别实现了接口中的函数）
type AddStu struct {
	x int
	y int
}

func (obj AddStu) funadd(val int) int {
	return obj.x + obj.y + val
}
func (obj AddStu) funmul() int {
	return obj.x * obj.y
}

```

## 调用

```
func RunUseInter(inter INTER) {

	//调用接口中的函数
	ret := inter.funadd(1000)
	fmt.Println(ret)
	ret = inter.funmul()
	fmt.Println(ret)
}


stu := AddStu{10, 15}
var inter INTER = stu //将结构体赋值给接口
RunUseInter(inter)
```
