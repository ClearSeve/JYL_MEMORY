# 数组

## 定义 
var n [4]int = [4]int{1, 2, 3, 4} //类型 [4]int

var n = make([]int, 1024)  //类型 []int

## 数组长度

len(aaa)

## 切片 
var a = n[2:]  //取第三个元素到末尾  
var b = n[:2]  //取头到第二个元素  
var c = n[1:3] //取第二到第三个元素

## 复制
aaa := []byte{0, 100, 200, 250}
bbb := []byte{1, 1, 1, 2, 3, 4}

//切片复制
//aaa的100~250拷贝到bbb的前三个1位置
copy(bbb[0:3], aaa[1:4])


## 遍历
```
for num := range n {
	fmt.Println(num)
}
```
