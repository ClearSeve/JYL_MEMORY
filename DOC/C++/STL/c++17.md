#include <charconv>
#include <string_view>
#include <system_error>

char pArray[10] = { 0 };
int valToStrShow = 999;
auto [pStr,err] = std::to_chars(pArray, pArray + 10, valToStrShow);
if (err == std::errc())
{
	int strlen = pStr - pArray;
	std::string_view sv(pArray, strlen);
	const char * str = sv.data();
	std::cout << "通过数字转换出的字符串:"<<str;
}
