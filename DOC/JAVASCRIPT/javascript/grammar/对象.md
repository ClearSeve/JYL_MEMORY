## 变量定义

```
var obj = {
	info: 'obj'
};

console.log(obj.info);//访问对象的成员
obj.fun = function(){
	console.log("define fun!!");
};//为对象增加成员
obj.fun();//访问对象的成员函数
```

## 使用函数构造

```
function People(name)
{
    this.name = name;
    this.show = function(age){alert(this.name + age);};
}


var people = new People("nn");
people.show(10);
```

## 声明后添加数据

```
var people = new Object();
people.name = "xx";
people.show = function(age){alert(this.name + age);};

people.show(12);
```
