

## 使用函数构造

```
function People(name)
{
    this.name = name;
    this.show = function(age){alert(this.name + age);};
}


var people = new People("nn");
people.show(10);
```

## 声明后添加数据

```
var people = new Object();
people.name = "xx";

//或使用
var people = {
	name: 'xx'
};


//为对象增加成员
people.show = function(age){alert(this.name + age);};
people.show(12);
```
