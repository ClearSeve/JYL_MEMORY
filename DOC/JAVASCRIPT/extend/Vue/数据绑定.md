# 数据绑定

## 文本绑定

```
<div id="divPos">
    {{ msg }}
</div>


<script>
const varDivPos = {
  data() {
    return {
        msg: 0
    }
  },
  mounted() {//启动事件
    setInterval(() => {
      this.msg++
    }, 1000)
  }
}
Vue.createApp(varDivPos).mount('#divPos')
</script>

```


## v-bind
```
<div id="divAttr">
    <span v-bind:title="msg">
        鼠标在位置上时显示时间
    </span>
</div>

const varDivAttr = {
  data() {
    return {
        msg:  new Date().toLocaleString()
    }
  }
}
Vue.createApp(varDivAttr).mount('#divAttr')
```


## v-on
```
<div id="divBtn">
  <p>{{ msg }}</p>
  <button v-on:click="fun">按钮</button>
</div>


const varDivBtn = {
  data() {
    return {
      msg: 'xxxx'
    }
  },
  methods: {
    fun() {
      this.msg = "yyyyy"
    }
  }
}

Vue.createApp(varDivBtn).mount('#divBtn')
```


## v-model
```
<div id="divM">
  <p>{{ msg }}</p>
  <input v-model="msg" />
</div>

const varDivM = {
  data() {
    return {
      msg: 'abc'
    }
  }
}
Vue.createApp(varDivM).mount('#divM')
```



## v-if
```
<div id="divIF">
    <span v-if="seen">xxxx</span>
</div>

const varDivIF = {
  data() {
    return {
      seen: false
    }
  }
}
Vue.createApp(varDivIF).mount('#divIF')
```

## v-for
```
<div id="divFOR">
  <ol>
    <li v-for="item in itemArryDat">
      {{ item.text }}
    </li>
</ol>
</div>


const varDivFOR = {
  data() {
    return {
    itemArryDat: [
      { text: '内容1' },
      { text: '内容2' },
      { text: '内容3' }
    ]
    }
  }
}

Vue.createApp(varDivFOR).mount('#divFOR')
```
