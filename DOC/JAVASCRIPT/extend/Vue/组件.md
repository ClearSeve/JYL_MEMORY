# 组件

```
<div id="divPos">
    <ol>
        <mycomponent v-for="item in itemArryDat" v-bind:myprops="item" v-bind:key="item.id"></mycomponent>
    </ol>
</div>

<script>
    Vue.component('mycomponent', {
        props: ['myprops'],
        template: '<li>{{ myprops.text }}</li>'
    })

    var posVar = new Vue({
        el: '#divPos',
        data: {
            itemArryDat: [
                { id: 0, text: '数据1' },
                { id: 1, text: '数据2' },
                { id: 2, text: '数据3' }
            ]
        }
    })
</script>
```
