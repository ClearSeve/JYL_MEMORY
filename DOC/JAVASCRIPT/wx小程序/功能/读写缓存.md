# 读写缓存

```
wx.setStorage({
  key: "key",
  data: "value",
  success:function(){
    wx.showToast({
      title: '保存成功',
      icon: 'success',
      duration: 3000
    });
  },
  fail: function() {
    wx.showToast({
      title: '保存失败',
      duration: 3000
    });
   }
});


wx.getStorage({
  key: 'key',
  success: function (res) {
    var data = res.data;
  },
  fail: function (res) { }
});
```
